Temperature & Humidity Monitor

Arduino Uno Project

Project Overview

To monitor environmental conditions and trigger alert if temperature exceeds safe threshold.

Components Used

  • Arduino Uno
  • DHT11
  • 220Ω Resistor
  • Breadboard
  • Jumper Wires
  • LED

How It Works

The DHT11 sends digital temperature & humidity data to the Arduino. The microcontrroller processes the data and activates an LED alert if the temperature crosses 30°C

Arduino Code


#include 

#define DHTPIN 2
#define DHTTYPE DHT11
#define ALERT_LED 8

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
  pinMode(ALERT_LED, OUTPUT);
  Serial.println("Environmental Monitoring System Started");
}

void loop() {
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();

  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("Sensor read error!");
    return;
  }

  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.print(" °C | Humidity: ");
  Serial.print(humidity);
  Serial.println(" %");

  // Temperature alert threshold
  if (temperature > 30) {
    digitalWrite(ALERT_LED, HIGH);
    Serial.println("WARNING: High Temperature!");
  } else {
    digitalWrite(ALERT_LED, LOW);
  }

  delay(2000);
}
